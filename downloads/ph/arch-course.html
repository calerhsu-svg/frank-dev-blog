Chapter 2: Singleton Pattern

public abstract class Singleton<T> : MonoBehaviour where T : MonoBehaviour
{
    private static T _instance;
    public static T Instance {
        get {
            if (_instance == null) {
                _instance = FindObjectOfType<T>();
                if (_instance == null) {
                    var go = new GameObject(typeof(T).Name);
                    _instance = go.AddComponent<T>();
                    DontDestroyOnLoad(go);
                }
            }
            return _instance;
        }
    }
}

public class AudioManager : Singleton<AudioManager> { }
⚠️ Don't overuse singletons. Only truly global systems need them (audio, save, network).

Chapter 3: Event System

public static class EventBus
{
    static Dictionary<string, Action<object>> events = new();

    public static void Subscribe(string name, Action<object> handler) {
        if (!events.ContainsKey(name)) events[name] = null;
        events[name] += handler;
    }

    public static void Publish(string name, object data = null) {
        if (events.ContainsKey(name)) events[name]?.Invoke(data);
    }
}

EventBus.Subscribe("player-died", (d) => UIManager.Instance.ShowGameOver());
EventBus.Publish("player-died");
💡 Event systems decouple module dependencies. Player doesn't need to know UIManager exists.

Chapter 4: State Machine (FSM)

public abstract class State {
    public virtual void Enter() { }
    public virtual void Update() { }
    public virtual void Exit() { }
}

public class StateMachine {
    private State current;
    public void ChangeState(State s) {
        current?.Exit(); current = s; current.Enter();
    }
    public void Update() => current?.Update();
}

Chapter 5: Object Pool

public class ObjectPool<T> where T : MonoBehaviour {
    Queue<T> pool = new();
    T prefab;

    public T Get() {
        T obj = pool.Count > 0 ? pool.Dequeue() : Object.Instantiate(prefab);
        obj.gameObject.SetActive(true);
        return obj;
    }

    public void Return(T obj) {
        obj.gameObject.SetActive(false);
        pool.Enqueue(obj);
    }
}
