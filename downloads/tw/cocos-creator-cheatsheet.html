📌 節點操作 API

// 取得節點
this.node                          // 當前節點
this.node.parent                   // 父節點
this.node.children                 // 子節點陣列
find("Canvas/Player")              // 全域路徑搜尋
this.node.getChildByName("Weapon") // 依名稱

// 位置 / 旋轉 / 縮放
this.node.setPosition(100, 200, 0)
this.node.getPosition()            // 回傳 Vec3
this.node.setRotationFromEuler(0, 0, 45)
this.node.setScale(2, 2, 1)

// 啟用 / 停用
this.node.active = true   // 顯示
this.node.active = false  // 隱藏（含子節點）

// 新增 / 移除子節點
parentNode.addChild(childNode)
childNode.removeFromParent()
this.node.destroyAllChildren()

🔄 生命週期回調

onLoad()      // 節點初始化（只一次）
              // 適合：取得 Component 引用

start()       // 第一次 update 前
              // 適合：初始化邏輯

onEnable()    // 節點啟用時
onDisable()   // 節點停用時

update(dt)    // 每幀呼叫
              // dt = delta time (秒)

lateUpdate(dt)// update 之後
              // 適合：攝影機跟隨

onDestroy()   // 節點銷毀時
              // 適合：清理事件監聽
⚡ 執行順序：onLoad → onEnable → start → update → lateUpdate

📡 事件系統

// 節點事件
this.node.on(Node.EventType.TOUCH_START, (e) => {
    console.log("觸摸開始", e.getUILocation())
}, this)

this.node.on(Node.EventType.TOUCH_MOVE, cb, this)
this.node.on(Node.EventType.TOUCH_END, cb, this)

// 自訂事件
this.node.emit("my-event", data)
this.node.on("my-event", (data) => {}, this)

// 全域事件（跨節點通訊）
director.on("game-over", handler, this)
director.emit("game-over", { score: 100 })

// 移除事件（防止記憶體洩漏！）
this.node.off(Node.EventType.TOUCH_START, cb, this)
director.off("game-over", handler, this)
⚠️ 務必在 onDestroy() 中移除監聽

✨ 動畫 & Tween

import { tween, Vec3, Quat } from 'cc'

// 基本 Tween
tween(this.node)
    .to(1, { position: new Vec3(100, 200, 0) })
    .start()

// 連續動畫
tween(this.node)
    .to(0.3, { scale: new Vec3(1.2, 1.2, 1) })
    .to(0.3, { scale: new Vec3(1, 1, 1) })
    .start()

// 重複 & 永久循環
tween(this.node)
    .by(1, { position: new Vec3(0, 20, 0) })
    .by(1, { position: new Vec3(0, -20, 0) })
    .union()
    .repeatForever()
    .start()

// 回呼
tween(this.node)
    .to(0.5, { opacity: 0 })
    .call(() => this.node.destroy())
    .start()

// 緩動函數
tween(this.node)
    .to(1, { position }, { easing: 'quadOut' })
    .start()
// 可用: linear, quadIn/Out, cubicIn/Out,
//       sineIn/Out, bounceOut, elasticOut

🔊 音效管理

import { AudioSource, resources, AudioClip } from 'cc'

// 掛在常駐節點上
let audio = this.getComponent(AudioSource)

// 播放 BGM
audio.clip = bgmClip
audio.loop = true
audio.volume = 0.5
audio.play()

// 播放 SFX（不中斷 BGM）
audio.playOneShot(sfxClip, 1.0)

// 動態載入音效
resources.load("audio/explosion", AudioClip,
    (err, clip) => {
        audio.playOneShot(clip)
    }
)

// 暫停 / 繼續
audio.pause()
audio.resume()

// 全域音量控制
AudioSource.maxAudioChannel = 24
