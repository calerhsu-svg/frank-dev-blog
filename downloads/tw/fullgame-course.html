第一章：遊戲企劃與原型設計

1.1 遊戲企劃文件 (GDD)

一頁企劃模板：
├── 遊戲名稱：Pixel Dungeon Rush
├── 類型：Roguelike 動作
├── 平台：iOS / Android
├── 目標受眾：休閒玩家 (18-35歲)
├── 核心玩法：一句話描述
│   「在隨機地牢中戰鬥、收集裝備、挑戰 Boss」
├── 營收模式：免費 + 廣告 + 內購
├── 開發週期：3 個月
└── 技術棧：Unity 2022.3 LTS + C#

1.2 快速原型

原型的目標是在最短時間內驗證核心玩法是否有趣。不要在原型階段花時間在美術、音效、UI 上。

💡 原型規則：48 小時內做出可玩版本。如果核心玩法不好玩，換方向比硬做更聰明。

第二章：核心玩法實作

// 角色控制器骨架
public class PlayerController : MonoBehaviour
{
    [Header("移動")]
    public float moveSpeed = 5f;
    public float dashSpeed = 15f;
    public float dashDuration = 0.2f;

    [Header("戰鬥")]
    public int attackDamage = 10;
    public float attackRange = 1.5f;
    public float attackCooldown = 0.5f;

    private Rigidbody2D rb;
    private StateMachine fsm;

    void Awake()
    {
        rb = GetComponent<Rigidbody2D>();
        fsm = new StateMachine();
        fsm.ChangeState(new IdleState(this));
    }

    void Update() => fsm.Update();
}

// 輸入處理（支援手機虛擬搖桿）
public class InputHandler : MonoBehaviour
{
    public Vector2 MoveInput { get; private set; }
    public bool AttackPressed { get; private set; }

    void Update()
    {
        // 鍵盤 + 手機雙支援
        MoveInput = new Vector2(
            Input.GetAxis("Horizontal"),
            Input.GetAxis("Vertical")
        );
        AttackPressed = Input.GetButtonDown("Fire1");
    }
}

// 數值平衡公式
// 敵人 HP = baseHP * (1 + level * 0.15)
// 敵人攻擊 = baseDMG * (1 + level * 0.1)
// 玩家預期 DPS = 需要的擊殺速度 * 敵人 HP
// 通關時間目標：每關 2-3 分鐘

public class UIManager : Singleton<UIManager>
{
    private Dictionary<string, UIPanel> panels = new();

    public void ShowPanel(string name)
    {
        if (panels.TryGetValue(name, out var panel))
            panel.Show();
    }

    public void HidePanel(string name)
    {
        if (panels.TryGetValue(name, out var panel))
            panel.Hide();
    }
}

第六章：存檔系統與雲端同步

// JSON 存檔系統
[System.Serializable]
public class SaveData
{
    public int level = 1;
    public int gold = 0;
    public int gems = 0;
    public List<string> unlockedItems = new();
    public string lastSaveTime;
}

public static class SaveSystem
{
    private static string path => Application.persistentDataPath + "/save.json";

    public static void Save(SaveData data)
    {
        data.lastSaveTime = DateTime.Now.ToString("o");
        string json = JsonUtility.ToJson(data, true);
        File.WriteAllText(path, json);
    }

    public static SaveData Load()
    {
        if (!File.Exists(path)) return new SaveData();
        string json = File.ReadAllText(path);
        return JsonUtility.FromJson<SaveData>(json);
    }
}

第八章：多語言與本地化

// 多語言系統
public class LocalizationManager : Singleton<LocalizationManager>
{
    private Dictionary<string, Dictionary<string, string>> translations;
    public string CurrentLang { get; private set; } = "zh-TW";

    public string Get(string key)
    {
        if (translations.TryGetValue(CurrentLang, out var dict))
            if (dict.TryGetValue(key, out var value))
                return value;
        return key; // fallback
    }
}

營運策略：
├── 第一週：Soft Launch（小區域測試）
├── 分析留存率：次日留存 > 40% 才正式上線
├── 第一個月：修 Bug、優化體驗
├── 每兩週更新：新關卡、活動、限時內容
└── ASO 優化：關鍵字、截圖、描述定期更新
