第一章：為什麼需要架構設計

當遊戲規模超過 1000 行程式碼，沒有架構的專案會快速變得難以維護。好的架構讓你能快速添加新功能、修復 Bug、和團隊協作。

常見的「義大利麵」問題

// ❌ 沒有架構：所有邏輯塞在一起
public class Player : MonoBehaviour {
    void Update() {
        HandleInput();
        HandleMovement();
        HandleAnimation();
        HandleUI();          // Player 為什麼要管 UI？
        HandleSound();       // Player 為什麼要管音效？
        HandleSaving();      // Player 為什麼要管存檔？
        HandleAchievements();// 成就系統也塞在這？
    }
}

// ✅ 好的架構：職責分離
// InputManager → 處理輸入
// PlayerController → 處理移動
// AnimationController → 處理動畫
// UIManager → 處理介面
// AudioManager → 處理音效
// SaveManager → 處理存檔

第二章：單例模式與全域管理器

// 安全的單例基底類別
public abstract class Singleton<T> : MonoBehaviour where T : MonoBehaviour
{
    private static T _instance;
    private static readonly object _lock = new object();

    public static T Instance
    {
        get
        {
            lock (_lock)
            {
                if (_instance == null)
                {
                    _instance = FindObjectOfType<T>();
                    if (_instance == null)
                    {
                        var go = new GameObject(typeof(T).Name);
                        _instance = go.AddComponent<T>();
                        DontDestroyOnLoad(go);
                    }
                }
                return _instance;
            }
        }
    }

    protected virtual void Awake()
    {
        if (_instance != null && _instance != this)
        {
            Destroy(gameObject);
            return;
        }
        _instance = this as T;
        DontDestroyOnLoad(gameObject);
    }
}

// 使用
public class AudioManager : Singleton<AudioManager> { }
// AudioManager.Instance.PlayBGM(clip);
⚠️ 單例不要過度使用。只有真正全域唯一的系統才用單例（音效、存檔、網路）。

第三章：觀察者模式與事件系統

// 簡易事件系統
public static class EventBus
{
    private static Dictionary<string, Action<object>> events = new();

    public static void Subscribe(string eventName, Action<object> handler)
    {
        if (!events.ContainsKey(eventName))
            events[eventName] = null;
        events[eventName] += handler;
    }

    public static void Unsubscribe(string eventName, Action<object> handler)
    {
        if (events.ContainsKey(eventName))
            events[eventName] -= handler;
    }

    public static void Publish(string eventName, object data = null)
    {
        if (events.ContainsKey(eventName))
            events[eventName]?.Invoke(data);
    }
}

// 使用範例
EventBus.Subscribe("player-died", (data) => {
    UIManager.Instance.ShowGameOver();
    AudioManager.Instance.PlaySFX(deathSound);
});

EventBus.Publish("player-died");
💡 事件系統解耦模組間的依賴。Player 不需要知道 UIManager 的存在。

第四章：狀態機模式 (FSM)

// 通用狀態機
public abstract class State
{
    public virtual void Enter() { }
    public virtual void Update() { }
    public virtual void Exit() { }
}

public class StateMachine
{
    private State currentState;

    public void ChangeState(State newState)
    {
        currentState?.Exit();
        currentState = newState;
        currentState.Enter();
    }

    public void Update() => currentState?.Update();
}

// 敵人 AI 狀態
public class PatrolState : State
{
    public override void Update()
    {
        // 巡邏邏輯
        if (CanSeePlayer())
            enemy.fsm.ChangeState(new ChaseState());
    }
}

public class ChaseState : State
{
    public override void Update()
    {
        // 追擊邏輯
        if (InAttackRange())
            enemy.fsm.ChangeState(new AttackState());
    }
}

第五章：物件池模式

public class ObjectPool<T> where T : MonoBehaviour
{
    private Queue<T> pool = new Queue<T>();
    private T prefab;
    private Transform parent;

    public ObjectPool(T prefab, int initialSize, Transform parent = null)
    {
        this.prefab = prefab;
        this.parent = parent;
        for (int i = 0; i < initialSize; i++)
            pool.Enqueue(CreateNew());
    }

    private T CreateNew()
    {
        var obj = Object.Instantiate(prefab, parent);
        obj.gameObject.SetActive(false);
        return obj;
    }

    public T Get()
    {
        T obj = pool.Count > 0 ? pool.Dequeue() : CreateNew();
        obj.gameObject.SetActive(true);
        return obj;
    }

    public void Return(T obj)
    {
        obj.gameObject.SetActive(false);
        pool.Enqueue(obj);
    }
}

// 使用：子彈池
ObjectPool<Bullet> bulletPool = new(bulletPrefab, 20);
Bullet b = bulletPool.Get();
// 子彈生命結束時
bulletPool.Return(b);

第六章：MVC/MVVM 在遊戲中的應用

// Model：純資料
public class PlayerModel
{
    public int HP { get; set; } = 100;
    public int MaxHP { get; set; } = 100;
    public int Score { get; set; } = 0;
    public event Action OnDataChanged;

    public void TakeDamage(int dmg)
    {
        HP = Mathf.Max(0, HP - dmg);
        OnDataChanged?.Invoke();
    }
}

// View：UI 顯示
public class PlayerView : MonoBehaviour
{
    public TextMeshProUGUI hpText;
    public Image hpBar;

    public void UpdateView(PlayerModel model)
    {
        hpText.text = $"{model.HP}/{model.MaxHP}";
        hpBar.fillAmount = (float)model.HP / model.MaxHP;
    }
}

// Controller：邏輯
public class PlayerController : MonoBehaviour
{
    PlayerModel model = new PlayerModel();
    PlayerView view;

    void Start()
    {
        view = GetComponent<PlayerView>();
        model.OnDataChanged += () => view.UpdateView(model);
    }
}

第七章：資源管理與配置系統

// ScriptableObject 配置系統
[CreateAssetMenu(fileName = "EnemyConfig", menuName = "Config/Enemy")]
public class EnemyConfig : ScriptableObject
{
    public string enemyName;
    public int hp;
    public float speed;
    public float attackDamage;
    public float attackRange;
    public GameObject prefab;
}

// 使用配置
public class Enemy : MonoBehaviour
{
    public EnemyConfig config;

    void Start()
    {
        hp = config.hp;
        speed = config.speed;
    }
}
