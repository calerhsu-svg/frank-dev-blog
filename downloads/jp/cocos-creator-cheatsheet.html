📌 ノード操作API

// ノード取得
this.node                          // 現在のノード
this.node.parent                   // 親ノード
this.node.children                 // 子ノード配列
find("Canvas/Player")              // グローバルパス検索
this.node.getChildByName("Weapon") // 名前で取得

// 位置 / 回転 / スケール
this.node.setPosition(100, 200, 0)
this.node.getPosition()            // Vec3を返す
this.node.setRotationFromEuler(0, 0, 45)
this.node.setScale(2, 2, 1)

// 有効 / 無効
this.node.active = true   // 表示
this.node.active = false  // 非表示

// 子ノード追加 / 削除
parentNode.addChild(childNode)
childNode.removeFromParent()

🔄 ライフサイクル

// 実行順序
onLoad()        // 最初に一度（初期化）
onEnable()      // 有効化時
start()         // 最初のupdate前に一度
update(dt)      // 毎フレーム
lateUpdate(dt)  // update後に毎フレーム
onDisable()     // 無効化時
onDestroy()     // 破棄時
💡 onLoadはノードのactiveに関係なく呼ばれます

スケジューラー

// タイマー
this.schedule(callback, interval, repeat, delay)
this.scheduleOnce(cb, 2)  // 2秒後に一度
this.unschedule(callback)
this.unscheduleAllCallbacks()

// ノードイベント

this.node.off(Node.EventType.TOUCH_START, this.onTouch, this)

Node.EventType.TOUCH_START

Node.EventType.TOUCH_END

// カスタムイベント

// グローバルイベント

✨ Tweenアニメーション

import { tween, Vec3 } from 'cc'

// 基本Tween
tween(this.node)
  .to(1, { position: new Vec3(100,200,0) })
  .start()

// チェーン
tween(this.node)
  .to(0.5, { scale: new Vec3(1.2,1.2,1) })
  .to(0.5, { scale: new Vec3(1,1,1) })
  .union()
  .repeatForever()
  .start()

// イージング
.to(1, { position: v }, { easing: 'sineOut' })

// コールバック
.call(() => { console.log('完了') })

⚡ 2D物理

import { RigidBody2D, PhysicsSystem2D } from 'cc'

// Rigidbody取得
const rb = this.getComponent(RigidBody2D)
rb.linearVelocity = new Vec2(5, 0)
rb.applyForceToCenter(new Vec2(0, 500), true)

// 衝突コールバック
onBeginContact(self, other, contact) {
  const tag = other.node.name
  if (tag === 'Enemy') this.takeDamage()
}

// Raycast
PhysicsSystem2D.instance.raycast(
  start, direction, type, mask
)
